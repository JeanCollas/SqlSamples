CREATE PROCEDURE [dbo].[Tech_SolveTransactionLogFull]
AS
BEGIN

	ALTER DATABASE MyDatabase
		SET RECOVERY SIMPLE

	DBCC SHRINKFILE(N'MyDatabase_log', 4096);Â 

	ALTER DATABASE MyDatabase
	    SET RECOVERY FULL
	
END
