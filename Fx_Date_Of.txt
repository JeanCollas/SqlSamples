-- =============================================

CREATE FUNCTION [dbo].[Month_Of](@date datetime)  
	RETURNS datetime
AS
BEGIN
	RETURN DATEFROMPARTS(DATEPART(year,@date),DATEPART(month,@date),DATEPART(day,@date))
END
